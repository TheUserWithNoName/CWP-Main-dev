is_western_bloc_member_loc = {
	# Western Bloc Member
	type = country
	text = {
		trigger = {
			or = {
				has_variable = western_bloc_member
				overlord = {
					or = {
						has_variable = western_bloc_member
						has_variable = western_bloc_leader
					}
				}
			}
		}
		localization_key = "yes"
	}
	text = {
		localization_key = "no"
	}
}
